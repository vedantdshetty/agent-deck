### Atomic write-temp-rename pattern — COPY VERBATIM from `internal/session/watcher_meta.go:47-75`

```go
func SaveState(name string, s *WatcherState) error {
    if name == "" {
        return fmt.Errorf("watcher name cannot be empty")
    }
    if s == nil {
        return fmt.Errorf("watcher state cannot be nil")
    }
    dir, err := WatcherDir(name)                           // from layout.go
    if err != nil {
        return err
    }
    if err := os.MkdirAll(dir, 0o755); err != nil {
        return fmt.Errorf("failed to create watcher dir: %w", err)
    }
    data, err := json.MarshalIndent(s, "", "  ")
    if err != nil {
        return fmt.Errorf("failed to marshal state.json: %w", err)
    }
    finalPath := filepath.Join(dir, "state.json")
    tmpPath := finalPath + ".tmp"
    if err := os.WriteFile(tmpPath, data, 0o644); err != nil {
        return fmt.Errorf("failed to write state.json.tmp: %w", err)
    }
    if err := os.Rename(tmpPath, finalPath); err != nil {
        _ = os.Remove(tmpPath)
        return fmt.Errorf("failed to rename state.json: %w", err)
    }
    return nil
}
```

### Migration guard — follow this shape in Task B

```go
// MigrateLegacyWatchersDir atomically renames ~/.agent-deck/watchers -> ~/.agent-deck/watcher
// on first boot, then creates a relative compatibility symlink watchers -> watcher.
// Single-shot; subsequent calls are no-ops. Uses os.Lstat (not os.Stat) to refuse
// a symlink-traversal attack where an adversary pre-creates ~/.agent-deck/watcher
// as a symlink outside the agent-deck root (threat T-21-SL).
func MigrateLegacyWatchersDir() error {
    deck, err := session.GetAgentDeckDir()
    if err != nil {
        return err
    }
    legacy := filepath.Join(deck, "watchers")
    current := filepath.Join(deck, "watcher")

    curInfo, curErr := os.Lstat(current)            // Lstat: do NOT follow symlink
    legacyInfo, legacyErr := os.Lstat(legacy)

    // SECURITY T-21-SL: if watcher/ already exists as a symlink with target outside the agent-deck dir, refuse.
    if curErr == nil && curInfo.Mode()&os.ModeSymlink != 0 {
        target, terr := os.Readlink(current)
        if terr == nil {
            resolved := target
            if !filepath.IsAbs(resolved) {
                resolved = filepath.Join(deck, resolved)
            }
            absDeck, _ := filepath.Abs(deck)
            absResolved, _ := filepath.Abs(resolved)
            if !strings.HasPrefix(absResolved, absDeck+string(os.PathSeparator)) && absResolved != absDeck {
                return fmt.Errorf("refusing migration: ~/.agent-deck/watcher is a symlink targeting %q (outside %s)", target, deck)
            }
        }
        // Symlink inside the deck — treat as "already migrated" (no-op).
        return nil
    }

    // Happy path: legacy exists as real dir, current missing -> rename + symlink.
    if legacyErr == nil && legacyInfo.Mode()&os.ModeSymlink == 0 && os.IsNotExist(curErr) {
        if err := os.Rename(legacy, current); err != nil {
            return fmt.Errorf("migrate watchers dir: %w", err)
        }
        if err := os.Symlink("watcher", legacy); err != nil {   // RELATIVE target
            slog.Warn("watcher: symlink creation failed (non-fatal)", "error", err)
        }
        slog.Info("watcher: migrated legacy ~/.agent-deck/watchers/ → ~/.agent-deck/watcher/",
            slog.String("note", "legacy ~/.agent-deck/issue-watcher/ NOT migrated (out of scope per REQ-WF-6)"))
        return nil
    }

    // Collision: both exist as real dirs — log and skip.
    if legacyErr == nil && legacyInfo.Mode()&os.ModeSymlink == 0 && curErr == nil && curInfo.Mode()&os.ModeSymlink == 0 {
        slog.Warn("watcher: both ~/.agent-deck/watchers/ and ~/.agent-deck/watcher/ exist; skipping migration")
        return nil
    }
    return nil
}
```

```go
func SaveState(name string, s *WatcherState) error {
    if name == "" {
        return fmt.Errorf("watcher name cannot be empty")
    }
    if s == nil {
        return fmt.Errorf("watcher state cannot be nil")
    }
    dir, err := WatcherDir(name)
    if err != nil {
        return err
    }
    if err := os.MkdirAll(dir, 0o755); err != nil {
        return fmt.Errorf("failed to create watcher dir: %w", err)
    }
    data, err := json.MarshalIndent(s, "", "  ")
    if err != nil {
        return fmt.Errorf("failed to marshal state.json: %w", err)
    }
    finalPath := filepath.Join(dir, "state.json")
    tmpPath := finalPath + ".tmp"
    if err := os.WriteFile(tmpPath, data, 0o644); err != nil {
        return fmt.Errorf("failed to write state.json.tmp: %w", err)
    }
    if err := os.Rename(tmpPath, finalPath); err != nil {
        _ = os.Remove(tmpPath)
        return fmt.Errorf("failed to rename state.json: %w", err)
    }
    return nil
}

// LoadState reads <name>/state.json. Returns (nil, nil) if the file does not exist (fresh watcher).
// Always hits disk — no in-process cache, so external edits are visible on next call (hot-reload safe).
func LoadState(name string) (*WatcherState, error) {
    dir, err := WatcherDir(name)
    if err != nil {
        return nil, err
    }
    statePath := filepath.Join(dir, "state.json")
    data, err := os.ReadFile(statePath)
    if err != nil {
        if errors.Is(err, fs.ErrNotExist) {
            return nil, nil
        }
        return nil, fmt.Errorf("failed to read state.json for watcher %q: %w", name, err)
    }
    var s WatcherState
    if err := json.Unmarshal(data, &s); err != nil {
        return nil, fmt.Errorf("failed to parse state.json for watcher %q: %w", name, err)
    }
    return &s, nil
}
```

```go
// Expected entry format: "## <RFC3339-ts> - <event_type>: <summary>" (no trailing newline — added here).

func AppendEventLog(name, entry string) error {
    dir, err := WatcherDir(name)

    if err != nil {
        return fmt.Errorf("failed to open task-log.md: %w", err)
    }
```

```go
func populateStateFields(e *watcherListEntry) {
    state, serr := watcher.LoadState(e.Name)
    if serr != nil || state == nil {
        e.LastEventTS = nil
        e.ErrorCount = 0
        e.HealthStatus = "unknown"
        return
    }
    ts := state.LastEventTS
    e.LastEventTS = &ts
    e.ErrorCount = state.ErrorCount
    e.HealthStatus = classifyFromState(state)
}

func classifyFromState(s *watcher.WatcherState) string {
    if !s.AdapterHealthy || s.ErrorCount >= 10 {
        return "error"
    }
    if s.ErrorCount >= 3 {
        return "warning"

}
```

| Boundary | Description |
|----------|-------------|
| Filesystem layout → agent-deck runtime | `~/.agent-deck/watcher/` and `~/.agent-deck/watchers/` paths are read on every engine boot. An adversary with write access to `~/.agent-deck/` (any local user process) can pre-create paths before first migration. |

| `state.json` / `meta.json` (user or external process writable) → engine state | Hot-reload reads disk on every `LoadState` call; a tampered file is surface for parsing bugs. Out of scope for threat disposition here (same trust boundary as existing `meta.json`). |
